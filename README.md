║                                                          ║
║ 🔍 Параметры поиска:                                     ║
║   country: ["US"]     — страна (US, RU, UA, KZ...)     ║
║   password: "no"      — без пароля                      ║
║   spam: "no"          — без спам-блока                  ║
║   not_origin: ["phishing"] — исключить фишинг          ║

║ 🚫 allowed_sellers   — покупка только у указанных      ║
